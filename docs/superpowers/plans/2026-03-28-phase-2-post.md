package com.community.domain.post.entity;

import lombok.Getter;

@Table(name = "categories")
@Getter

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    /** 표시명 (예: "자유") */
    @Column(nullable = false, unique = true, length = 50)

    /** URL용 슬러그 (예: "free") */

    private String description;

    @Builder.Default
    private int sortOrder = 0;

    @Column(nullable = false)
    @Builder.Default
    private boolean isActive = true;

    @Column(nullable = false, updatable = false)
    @Builder.Default
    private LocalDateTime createdAt = LocalDateTime.now();
}

 * DB 테이블: posts (V3 마이그레이션)

package com.community.domain.post.entity;

import com.community.domain.user.entity.User;
import com.community.global.common.entity.BaseTimeEntity;
import jakarta.persistence.*;
import lombok.AccessLevel;
import lombok.Builder;
import lombok.Getter;
import lombok.NoArgsConstructor;

@Entity
@Table(name = "posts")
@Getter
@NoArgsConstructor(access = AccessLevel.PROTECTED)
@Builder
public class Post extends BaseTimeEntity {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    /** 게시글 작성자 — LAZY 로딩으로 N+1 방지 */
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "author_id", nullable = false)
    private User author;

    /** 카테고리 삭제 시 NULL 유지 */
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "category_id")
    private Category category;

    @Column(nullable = false, length = 200)
    private String title;

    /** TipTap 에디터의 HTML 출력 */
    @Column(nullable = false, columnDefinition = "TEXT")
    private String content;

    /** 목록 미리보기용 — HTML 태그 제거 후 첫 300자 */
    @Column(length = 300)
    private String summary;

    @Column(length = 500)
    private String thumbnailUrl;

    @Column(nullable = false)
    @Builder.Default
    private int viewCount = 0;

    /** 비정규화: 매번 COUNT 쿼리 방지 */
    @Column(nullable = false)
    @Builder.Default
    private int voteCount = 0;

    @Column(nullable = false)
    @Builder.Default
    private int commentCount = 0;

    @Column(nullable = false)
    @Builder.Default
    private boolean isPinned = false;

    /** Soft Delete */
    @Column(nullable = false)
    @Builder.Default
    private boolean isDeleted = false;

    // ── 비즈니스 메서드 ──

    /** 게시글 내용 수정 (null이면 기존값 유지) */
    public void update(String title, String content, String summary, Category category) {
        if (title != null) this.title = title;
        if (content != null) {
            this.content = content;
            this.summary = summary;
        }
        if (category != null) this.category = category;
    }

    public void softDelete() { this.isDeleted = true; }
    public void pin() { this.isPinned = true; }
    public void unpin() { this.isPinned = false; }
    public void increaseViewCount() { this.viewCount++; }
    public void increaseVoteCount() { this.voteCount++; }
    public void decreaseVoteCount() { if (this.voteCount > 0) this.voteCount--; }
    public void increaseCommentCount() { this.commentCount++; }
    public void decreaseCommentCount() { if (this.commentCount > 0) this.commentCount--; }
}

- [ ] **Step 4: PostRepositoryImpl.java 작성**

```java
/**
 * 게시글 QueryDSL 검색 구현체
 * author를 fetchJoin하여 N+1 방지. isDeleted=false 항상 적용.
 */
package com.community.domain.post.repository;

import com.community.domain.post.entity.Post;
import com.community.domain.post.entity.QPost;
import com.community.domain.user.entity.QUser;
import com.querydsl.core.types.OrderSpecifier;
import com.querydsl.jpa.impl.JPAQueryFactory;
import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.PageImpl;
import org.springframework.data.domain.Pageable;
import org.springframework.stereotype.Repository;
import java.util.List;

@Repository
@RequiredArgsConstructor
public class PostRepositoryImpl implements PostRepositoryCustom {

    private final JPAQueryFactory queryFactory;

    @Override
    public Page<Post> search(String keyword, Long categoryId, String sort, Pageable pageable) {
        QPost post = QPost.post;
        QUser user = QUser.user;

        OrderSpecifier<?> orderSpecifier = "POPULAR".equals(sort)
                ? post.voteCount.desc()
                : post.createdAt.desc();

        List<Post> content = queryFactory
                .selectFrom(post)
                .join(post.author, user).fetchJoin()
                .where(
                        post.isDeleted.isFalse(),
                        keyword != null ? post.title.containsIgnoreCase(keyword) : null,
                        categoryId != null ? post.category.id.eq(categoryId) : null
                )
                .orderBy(orderSpecifier)
                .offset(pageable.getOffset())
                .limit(pageable.getPageSize())
                .fetch();

        long total = queryFactory
                .select(post.count())
                .from(post)
                .where(
                        post.isDeleted.isFalse(),
                        keyword != null ? post.title.containsIgnoreCase(keyword) : null,
                        categoryId != null ? post.category.id.eq(categoryId) : null
                )
                .fetchOne();

        return new PageImpl<>(content, pageable, total);
    }
}
```

/**
 * 게시글 서비스
 *
 * 게시글 CRUD 비즈니스 로직. HTML content에서 태그를 제거해 summary를 추출합니다.
 * 읽기: @Transactional(readOnly=true), 쓰기: @Transactional
 *
 * 의존성: PostRepository, CategoryRepository, UserRepository, PostMapper
 * 사용 위치: PostController
 */
package com.community.domain.post.service;

import com.community.domain.post.dto.request.CreatePostRequest;
import com.community.domain.post.dto.request.PostSearchRequest;
import com.community.domain.post.dto.request.UpdatePostRequest;
import com.community.domain.post.dto.response.PostDetailResponse;
import com.community.domain.post.dto.response.PostListResponse;
import com.community.domain.post.entity.Category;
import com.community.domain.post.entity.Post;
import com.community.domain.post.mapper.PostMapper;
import com.community.domain.post.repository.CategoryRepository;
import com.community.domain.post.repository.PostRepository;
import com.community.domain.user.entity.User;
import com.community.domain.user.entity.UserRole;
import com.community.domain.user.repository.UserRepository;
import com.community.global.common.exception.BusinessException;
import com.community.global.common.response.ErrorCode;
import com.community.global.common.response.PageResponse;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.PageRequest;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.util.List;

@Service
@RequiredArgsConstructor
@Slf4j
public class PostService {

    private final PostRepository postRepository;
    private final CategoryRepository categoryRepository;
    private final UserRepository userRepository;
    private final PostMapper postMapper;

    /** 게시글 목록 조회 (페이징 + 동적 필터) */
    @Transactional(readOnly = true)
    public PageResponse<PostListResponse> getPosts(PostSearchRequest req) {
        PageRequest pageable = PageRequest.of(req.page(), req.size());
        Page<Post> page = postRepository.search(req.keyword(), req.categoryId(), req.sort(), pageable);
        return PageResponse.of(page.map(postMapper::toListResponse));
    }

    /** 게시글 상세 조회 + 조회수 증가 */
    @Transactional
    public PostDetailResponse getPost(Long postId, Long userId) {
        Post post = postRepository.findByIdAndIsDeletedFalse(postId)
                .orElseThrow(() -> new BusinessException(ErrorCode.POST_NOT_FOUND));
        post.increaseViewCount();
        return postMapper.toDetailResponse(post);
    }

    /** 게시글 작성 */
    @Transactional
    public PostDetailResponse createPost(Long userId, CreatePostRequest req) {
        User author = userRepository.findById(userId)
                .orElseThrow(() -> new BusinessException(ErrorCode.USER_NOT_FOUND));

        Category category = null;
        if (req.categoryId() != null) {
            category = categoryRepository.findById(req.categoryId())
                    .orElseThrow(() -> new BusinessException(ErrorCode.CATEGORY_NOT_FOUND));
        }

        Post post = Post.builder()
                .author(author)
                .category(category)
                .title(req.title())
                .content(req.content())
                .summary(extractSummary(req.content()))
                .thumbnailUrl(req.thumbnailUrl())
                .build();

        return postMapper.toDetailResponse(postRepository.save(post));
    }

    /** 게시글 수정 (작성자 본인만) */
    @Transactional
    public PostDetailResponse updatePost(Long postId, Long userId, UpdatePostRequest req) {
        Post post = postRepository.findByIdAndIsDeletedFalse(postId)
                .orElseThrow(() -> new BusinessException(ErrorCode.POST_NOT_FOUND));

        if (!post.getAuthor().getId().equals(userId)) {
            throw new BusinessException(ErrorCode.UNAUTHORIZED_MODIFICATION);
        }

        Category category = null;
        if (req.categoryId() != null) {
            category = categoryRepository.findById(req.categoryId())
                    .orElseThrow(() -> new BusinessException(ErrorCode.CATEGORY_NOT_FOUND));
        }

        String newSummary = req.content() != null ? extractSummary(req.content()) : null;
        post.update(req.title(), req.content(), newSummary, category);
        return postMapper.toDetailResponse(post);
    }

    /** 게시글 삭제 (작성자 또는 ADMIN, Soft Delete) */
    @Transactional
    public void deletePost(Long postId, Long userId) {
        Post post = postRepository.findByIdAndIsDeletedFalse(postId)
                .orElseThrow(() -> new BusinessException(ErrorCode.POST_NOT_FOUND));

        User requestUser = userRepository.findById(userId)
                .orElseThrow(() -> new BusinessException(ErrorCode.USER_NOT_FOUND));

        boolean isAuthor = post.getAuthor().getId().equals(userId);
        boolean isAdmin = requestUser.getRole() == UserRole.ADMIN;

        if (!isAuthor && !isAdmin) {
            throw new BusinessException(ErrorCode.UNAUTHORIZED_MODIFICATION);
        }

        post.softDelete();
    }

    /**
     * HTML 태그 제거 후 summary 추출 (최대 300자)
     * TipTap 에디터 출력은 HTML이므로 정규식으로 태그를 제거합니다.
     */
    private String extractSummary(String htmlContent) {
        if (htmlContent == null) return null;
        String text = htmlContent
                .replaceAll("<[^>]+>", "")
                .replaceAll("&nbsp;", " ")
                .replaceAll("&lt;", "<")
                .replaceAll("&gt;", ">")
                .replaceAll("&amp;", "&")
                .replaceAll("\\s+", " ")
                .trim();
        return text.length() > 300 ? text.substring(0, 300) : text;
    }

    /** 활성 카테고리 목록 조회 */
    @Transactional(readOnly = true)
    public List<Category> getActiveCategories() {
        return categoryRepository.findAllByIsActiveTrueOrderBySortOrder();
    }
}

/**
 * 무한 스크롤 IntersectionObserver 훅
 * 반환된 ref를 목록 끝 요소에 붙이면 뷰포트 진입 시 onIntersect 호출
 * 사용 위치: PostList
 */

import { useEffect, useRef } from 'react';

interface UseInfiniteScrollOptions {
  onIntersect: () => void;
  rootMargin?: string;
  enabled?: boolean;
}

export function useInfiniteScroll({ onIntersect, rootMargin = '200px', enabled = true }: UseInfiniteScrollOptions) {
  const targetRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!enabled) return;
    const target = targetRef.current;
    if (!target) return;
    const observer = new IntersectionObserver(
      (entries) => { if (entries[0].isIntersecting) onIntersect(); },
      { rootMargin },
    );
    observer.observe(target);
    return () => observer.disconnect();
  }, [onIntersect, rootMargin, enabled]);

  return targetRef;
}

    return d.format('YYYY.MM.DD');
  }, [date]);

  return (
    <Typography variant="caption" color="text.secondary" {...typographyProps}>
      {label}
    </Typography>
  );

```typescript
/**
 * 게시글 작성/수정 페이지
 * :id 유무로 작성/수정 모드 구분. 이탈 시 확인 다이얼로그.
 * summary는 content HTML 태그 제거 후 첫 300자 자동 추출.
 * 사용 위치: router.tsx /posts/new, /posts/:id/edit
 */

import { useState, useEffect } from 'react';
import Box from '@mui/material/Box';
import TextField from '@mui/material/TextField';
import Select from '@mui/material/Select';
import MenuItem from '@mui/material/MenuItem';
import FormControl from '@mui/material/FormControl';
import InputLabel from '@mui/material/InputLabel';
import Button from '@mui/material/Button';
import Typography from '@mui/material/Typography';
import Dialog from '@mui/material/Dialog';
import DialogTitle from '@mui/material/DialogTitle';
import DialogContent from '@mui/material/DialogContent';
import DialogContentText from '@mui/material/DialogContentText';
import DialogActions from '@mui/material/DialogActions';
import { useParams, useNavigate } from 'react-router';
import { PostEditor } from '../components/PostEditor';
import { usePost } from '../hooks/usePost';
import { useCreatePost, useUpdatePost } from '../hooks/usePostMutation';
import { useCategories } from '../hooks/usePosts';
import { tokens } from '@/app/theme/tokens';

function extractSummary(html: string): string {
  return html.replace(/<[^>]+>/g, '').replace(/&nbsp;/g, ' ').replace(/\s+/g, ' ').trim().slice(0, 300);
}

export function PostEditorPage() {
  const { id } = useParams<{ id: string }>();
  const navigate = useNavigate();
  const isEditMode = !!id;
  const postId = id ? Number(id) : undefined;

  const { data: existingPost } = usePost(postId);
  const { data: categories = [] } = useCategories();
  const { mutate: createPost, isPending: isCreating } = useCreatePost();
  const { mutate: updatePost, isPending: isUpdating } = useUpdatePost(postId!);

  const [title, setTitle] = useState('');
  const [content, setContent] = useState('');
  const [categoryId, setCategoryId] = useState<number | ''>('');
  const [leaveDialogOpen, setLeaveDialogOpen] = useState(false);
  const [isDirty, setIsDirty] = useState(false);

  useEffect(() => {
    if (existingPost && isEditMode) {
      setTitle(existingPost.title);
      setContent(existingPost.content);
      if (existingPost.categoryName) {
        const cat = categories.find((c) => c.name === existingPost.categoryName);
        if (cat) setCategoryId(cat.id);
      }
    }
  }, [existingPost, isEditMode, categories]);

  const handleSubmit = () => {
    const payload = { title, content, categoryId: categoryId || undefined, summary: extractSummary(content) };
    if (isEditMode) { updatePost(payload); } else { createPost(payload); }
  };

  return (
    <Box sx={{ p: tokens.spacing.page, maxWidth: 800, mx: 'auto' }}>
      <Box sx={{ display: 'flex', alignItems: 'center', mb: 3, gap: 2 }}>
        <Button variant="text" onClick={() => { if (isDirty) setLeaveDialogOpen(true); else navigate(-1); }}>뒤로가기</Button>
        <Typography variant="h6" sx={{ flex: 1 }}>{isEditMode ? '게시글 수정' : '게시글 작성'}</Typography>
        <Button variant="contained" onClick={handleSubmit} disabled={!title.trim() || !content.trim() || isCreating || isUpdating}>
          {isEditMode ? '수정하기' : '게시하기'}
        </Button>
      </Box>

      <FormControl fullWidth sx={{ mb: 2 }}>
        <InputLabel>카테고리 (선택)</InputLabel>
        <Select value={categoryId} label="카테고리 (선택)" onChange={(e) => setCategoryId(e.target.value as number | '')}>
          <MenuItem value="">선택 안함</MenuItem>
          {categories.map((cat) => <MenuItem key={cat.id} value={cat.id}>{cat.name}</MenuItem>)}
        </Select>
      </FormControl>

      <TextField fullWidth label="제목" value={title} onChange={(e) => { setTitle(e.target.value); setIsDirty(true); }} inputProps={{ maxLength: 200 }} sx={{ mb: 2 }} />

      <PostEditor content={content} onChange={(html) => { setContent(html); setIsDirty(true); }} />

      <Dialog open={leaveDialogOpen} onClose={() => setLeaveDialogOpen(false)}>
        <DialogTitle>작성 취소</DialogTitle>
        <DialogContent><DialogContentText>작성을 취소하시겠습니까? 입력한 내용이 사라집니다.</DialogContentText></DialogContent>
        <DialogActions>
          <Button onClick={() => setLeaveDialogOpen(false)}>계속 작성</Button>
          <Button color="error" onClick={() => navigate(-1)}>취소</Button>
        </DialogActions>
      </Dialog>
    </Box>
  );
}
```
